import gymnasium as gym
import jax
import jax.numpy as jnp

# 1. Policy (décision)
def policy(params, obs):
    return jnp.dot(obs, params['w']) + params['b']

# 2. Loss (apprentissage)
def compute_loss(params, trajectories):
    loss = 0
    for obs, action, reward in trajectories:
        logits = policy(params, obs)
        log_prob = jax.nn.log_softmax(logits)[action]
        loss -= log_prob * reward  # REINFORCE
    return loss / len(trajectories)

# 3. Update (optimisation)
@jax.jit
def update(params, trajectories, lr=0.01):
    grads = jax.grad(compute_loss)(params, trajectories)
    return {k: params[k] - lr * grads[k] for k in params}

# 4. Collect (interaction)
def collect_episode(env, params, rng):
    obs, _ = env.reset()
    done = False
    traj = []
    while not done:
        rng, _rng = jax.random.split(rng)
        action = jax.random.categorical(_rng, policy(params, obs))
        next_obs, reward, terminated, truncated, _ = env.step(int(action))
        done = terminated or truncated
        traj.append((obs, action, reward))
        obs = next_obs
    return traj

# 5. Evaluate (test)
def evaluate(env, params, n_episodes=5):
    total = 0
    for _ in range(n_episodes):
        traj = collect_episode(env, params, jax.random.PRNGKey(0))
        total += sum(r for _, _, r in traj)
    return total / n_episodes

# Boucle principale
env = gym.make("CartPole-v1")
rng = jax.random.PRNGKey(42)
params = {'w': jax.random.normal(rng, (4, 2)) * 0.1,
          'b': jax.random.normal(rng, (2,)) * 0.1}

for episode in range(100):
    rng, _rng = jax.random.split(rng)
    trajectories = collect_episode(env, params, _rng)
    params = update(params, trajectories)
    
    if episode % 20 == 0:
        score = evaluate(env, params)
        print(f"Épisode {episode}, Score moyen: {score}")
